{% trans "FHA Calculation Formulas" %}

{% trans "Upfront MIP Formula" %}

UFMIP = Loan Amount × 1.75%

{% trans "Example: $350,000 × 0.0175 = $6,125" %}

{% trans "Monthly MIP Formula" %}

Monthly MIP = (Loan × Annual Rate) ÷ 12

{% trans "Example: ($356,125 × 0.55%) ÷ 12 = $163" %}
