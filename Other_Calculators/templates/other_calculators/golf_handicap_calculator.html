{% trans "Key Golf Handicap Formulas" %}

{% trans "Handicap Index (WHS):" %}

HI = Average of best differentials × 0.96

{% trans "Course Handicap:" %}

CH = (HI × SR / 113) + (CR - Par)

{% trans "Score Differential:" %}

SD = (AGS - CR) × (113 / SR)

{% trans "Net Score:" %}

Net = Gross - Course Handicap

{% trans "Playing Handicap:" %}

PH = CH × (Allowance / 100)

{% trans "Where:" %}

  • HI = {% trans "Handicap Index" %}
  • CH = {% trans "Course Handicap" %}
  • SR = {% trans "Slope Rating" %}
  • CR = {% trans "Course Rating" %}
  • AGS = {% trans "Adjusted Gross Score" %}
  • 113 = {% trans "Standard Slope Rating" %}
