{% trans "GPA Formula" %}

{% trans "Semester GPA:" %}

GPA = Total Quality Points / Total Credits

{% trans "Quality Points:" %}

Quality Points = Grade Points × Credits

{% trans "Cumulative GPA:" %}

Cumulative GPA = (Previous Points + New Points) / (Previous Credits + New Credits)

{% trans "Where:" %}

  • {% trans "Grade Points = Points assigned to letter grade (A=4.0, B=3.0, etc.)" %}
  • {% trans "Credits = Credit hours for the course" %}
  • {% trans "Quality Points = Grade Points × Credits" %}
