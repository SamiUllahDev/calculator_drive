{% trans "Key Subnet Formulas" %}

{% trans "Network Address:" %}

Network Address = IP Address AND Subnet Mask

{% trans "Broadcast Address:" %}

Broadcast Address = Network Address OR (NOT Subnet Mask)

{% trans "Total Hosts:" %}

Total Hosts = 2^(32 - CIDR)

{% trans "Usable Hosts:" %}

Usable Hosts = Total Hosts - 2 (network + broadcast)

{% trans "Subnet from Hosts:" %}

CIDR = 32 - ⌈log2(Hosts Needed + 2)⌉

{% trans "Subnet from Subnets:" %}

New CIDR = Original CIDR + ⌈log2(Subnets Needed)⌉
