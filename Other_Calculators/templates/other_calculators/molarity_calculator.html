{% trans "Formulas" %}

{% trans "Molarity" %}:
M = n / V

{% trans "Moles" %}:
n = M × V

{% trans "Volume" %}:
V = n / M

{% trans "Moles from Mass" %}:
n = m / MW

{% trans "Mass" %}:
m = n × MW

{% trans "Dilution" %}:
