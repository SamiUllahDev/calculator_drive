{% trans "Formulas" %}

{% trans "Sidewall Height" %}:

Sidewall = (Width × Aspect Ratio / 100) / 25.4

{% trans "Overall Diameter" %}:

Diameter = Rim + (2 × Sidewall)

{% trans "Circumference" %}:

Circumference = π × Diameter

{% trans "Revolutions per Mile" %}:

Rev/Mile = 63360 / Circumference
